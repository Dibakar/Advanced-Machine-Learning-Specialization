import tensorflow as tf
s = tf.InteractiveSession()
import numpy as np
from sklearn.metrics import roc_auc_score

X_train = X_train.reshape([-1, 28*28])
y_train_one_hot = np.zeros((y_train.shape[0], 10), dtype='float32')
y_train_one_hot[np.arange(y_train.shape[0]), y_train] = 1
X_val = X_val.reshape([-1, 28*28])
y_val_one_hot = np.zeros((y_val.shape[0], 10), dtype='float32')
y_val_one_hot[np.arange(y_val.shape[0]), y_val] = 1
X_test = X_test.reshape([-1, 28*28])
y_test_one_hot = np.zeros((y_test.shape[0], 10), dtype='float32')
y_test_one_hot[np.arange(y_test.shape[0]), y_test] = 1


# Model parameters - weights and bias (Network with one hidden layer, one output layer)
weights_hidden_1 = tf.Variable(tf.truncated_normal([X_train.shape[1], 50], stddev=0.001), name="hidden_layer_1_weights")
b_hidden_1 = tf.Variable(tf.zeros([50]), name="hidden_layer_1_biases")
weights_output = tf.Variable(tf.truncated_normal([50,10], stddev=0.001), name="output_layer_weights")
b_output = tf.Variable(tf.zeros([10]), name="output_layer_biases")

# Placeholders for the input data
input_X = tf.placeholder(tf.float32, [None,X_train.shape[1]])
input_y = tf.placeholder(tf.float32, [None,10])

# Network with one hidden layer, one output layer
hidden_layer_1 = tf.nn.sigmoid(tf.matmul(input_X, weights_hidden_1) + b_hidden_1)
predicted_y = tf.matmul(hidden_layer_1, weights_output) + b_output

#Cross-entropy loss
loss = tf.reduce_mean(tf.nn.softmax_cross_entropy_with_logits(labels=input_y, logits=predicted_y))

#<optimizer that minimizes loss>
#optimizer = tf.train.GradientDescentOptimizer(0.5).minimize(loss)
optimizer = tf.train.MomentumOptimizer(0.5, 0.8).minimize(loss) #<optimizer that minimizes loss>

#Compute accuracy
prediction = tf.argmax(predicted_y, 1)
correct_answer = tf.placeholder(tf.int64, [None])
equality = tf.equal(prediction, correct_answer)
accuracy = tf.reduce_mean(tf.cast(equality, tf.float32))

# keep results for plotting
train_loss_results = []
test_loss_results = []
iteration_numbers = []

#Train the network using mini-batch gradient descent with batch size = 128
s.run(tf.global_variables_initializer())
for i in range(2000):
    sample_index = np.random.randint(X_train.shape[0], size=128) #using random subsamples on each iteration, mini-batch size=128 
    s.run(optimizer, {input_X: X_train[sample_index], input_y: y_train_one_hot[sample_index]})
    train_loss = s.run(loss, {input_X: X_train[sample_index], input_y: y_train_one_hot[sample_index]})
    test_loss = s.run(loss, {input_X: X_test, input_y: y_test_one_hot})
    train_loss_results.append(train_loss)
    test_loss_results.append(test_loss)
    iteration_numbers.append(i)
    if (i%100 == 0): print("Train loss, test loss at iter %i: %.4f, %.4f" % (i, train_loss, test_loss))
    if (i%100 == 0): print ("Accuracy on test data at iter %i: %.4f" % (i, s.run(accuracy, {input_X:X_test, correct_answer: y_test})))

#Evaluate the network on test dataset
print ("\n")
print ("Final accuracy on test dataset: %.4f" % s.run(accuracy, {input_X:X_test, correct_answer: y_test}))
print ("\n")

#plot train loss and test losses as a function of the iteration number
fig, axes = plt.subplots(2, sharex=True, figsize=(12, 8))

axes[0].set_ylabel("Train loss", fontsize=14)
axes[0].set_xlabel("Iteration number", fontsize=14)
axes[0].plot(iteration_numbers, train_loss_results)

axes[1].set_ylabel("Test loss", fontsize=14)
axes[1].set_xlabel("Iteration number", fontsize=14)
axes[1].plot(iteration_numbers, test_loss_results)
